#User
User Signup
     path : api/user/signup
     method : post
     Parameters: 
                           * email              		  -  (mandatory)
                           * password                     -  (mandatory)
                 	       * firstname                    -  (mandatory)
                           * lastname                	  -  (mandatory)
                           * address                      -  (mandatory)

 response:
              	* for error   : ({ error: 'signup failed', error_description: 'error' })
              	* for success : (({ message: 'signup successful'})

User login
     path : api/user/login
     method : post
     Parameters: 
                           * email              		  -  (mandatory)
                           * password                     -  (mandatory)
 response:
              * for error   : ({ error: 'login failed', error_description: "incorrect password! })
              * for success : (({ message: 'login successful'})
